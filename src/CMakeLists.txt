set(QUERYDEF_PL "${CMAKE_SOURCE_DIR}/src/querydef.pl")
add_custom_command(OUTPUT querydef.h
	COMMAND ${QUERYDEF_PL}
	DEPENDS ${QUERYDEF_PL}
	COMMENT "Generating querydef.h..."
	WORKING_DIRECTORY "${CMAKE_BINARY_DIR}/src")

	WORKING_DIRECTORY "${CMAKE_BINARY_DIR}/src")
